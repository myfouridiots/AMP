//MainActivity.kt

package com.example.intent_239744
import android.content.Intent
import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.widget.Button
import android.widget.EditText

class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val name = findViewById<EditText>(R.id.name)
        val rollnum = findViewById<EditText>(R.id.rollnum)
        val age = findViewById<EditText>(R.id.age)
        val btn = findViewById<Button>(R.id.btn)

        btn.setOnClickListener{
            val nameString = name.text.toString().trim()
            val rollnumString = rollnum.text.toString().trim()
            val ageString = age.text.toString().trim()
            val intent = Intent(this, SecondActivity::class.java)
                intent.putExtra("name", nameString)
                intent.putExtra("rollnum", rollnumString)
                intent.putExtra("age", ageString)
                startActivity(intent)

        }
    }
}
